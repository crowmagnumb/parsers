18  public class TextDateParserTest {
19  
20    private TextDateParser TEXTDATE_PARSER = new TextDateParser();
21  
22    @Test
23    public void testTextDateParsing(){
24      ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("2nd jan. 2018");
25      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
26  
27      parseResult = TEXTDATE_PARSER.parse("2018 january 2");
28      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
29  
30      parseResult = TEXTDATE_PARSER.parse("2018-01-02");
31      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
32  
33      parseResult = TEXTDATE_PARSER.parse("2018年1月2日");
34      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
35    }
36  
37    @Test
38    public void testTextDateTimeParsing(){
39      ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("1978-1-1T00:00");
40      assertEquals(LocalDate.of(1978, Month.JANUARY, 1), LocalDate.from(parseResult.getPayload()));
41  
42      parseResult = TEXTDATE_PARSER.parse("1978-01-01T02:12:43Z");
43      assertEquals(LocalDate.of(1978, Month.JANUARY, 1), LocalDate.from(parseResult.getPayload()));
44    }
45  
46    @Test
47    public void testDateParts(){
48      ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("2018.0", "1.0", "02.0");
49      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
50  
51      parseResult = TEXTDATE_PARSER.parse("2018.0", "jan", "02.0");
52      assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
53  
54      parseResult = TEXTDATE_PARSER.parse("2018.0", "jan", "");
55      assertEquals(YearMonth.of(2018, Month.JANUARY), YearMonth.from(parseResult.getPayload()));
56    }
57  
58    @Test
59    public void testTextDateParsingNullEmpty(){
60      ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("");
61      assertFalse(parseResult.isSuccessful());
62  
63      parseResult = TEXTDATE_PARSER.parse(null);
64      assertFalse(parseResult.isSuccessful());
65    }
66  
67    @Test
68    public void testTextDateParsingInvalidDate(){
69      ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("32nd jan. 2018");
70      assertFalse(parseResult.isSuccessful());
71  
72      parseResult = TEXTDATE_PARSER.parse("2nd n/a 2018");
73      assertFalse(parseResult.isSuccessful());
74  
75      parseResult = TEXTDATE_PARSER.parse("15 jan-fev 2018");
76      assertFalse(parseResult.isSuccessful());
77    }
78  
79    @Test
80    public void testTextDatePartsParsing(){
81      ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("1879", "11", "1");
82      assertEquals(LocalDate.of(1879, Month.NOVEMBER, 1), LocalDate.from(parseResult.getPayload()));
83  
84      parseResult = TEXTDATE_PARSER.parse("1879", "a", "1");
85      assertFalse(parseResult.isSuccessful());
86    }
