001package org.gbif.nameparser;
002
003import org.gbif.api.vocabulary.Rank;
004
005import java.util.HashMap;
006import java.util.Map;
007
008import org.junit.Test;
009
010import static org.junit.Assert.assertEquals;
011
012/**
013 *
014 */
015public class RankUtilsTest {
016  private static final Map<String, Rank> NAMES = new HashMap<String, Rank>();
017
018  static {
019    NAMES.put("Asteraceae", Rank.FAMILY);
020    NAMES.put("Magnoliophyta", Rank.PHYLUM);
021    NAMES.put("Fabales", Rank.ORDER);
022    NAMES.put("Hominidae", Rank.FAMILY);
023    NAMES.put("Drosophilinae", Rank.SUBFAMILY);
024    NAMES.put("Agaricomycetes", Rank.CLASS);
025  }
026  
027  @Test
028  public void testInferRank() {
029    for (Map.Entry<String, Rank> stringRankEntry : NAMES.entrySet()) {
030      assertEquals(stringRankEntry.getValue(), RankUtils.inferRank(stringRankEntry.getKey(), null, null, null, null));
031    }
032
033    assertEquals(Rank.SPECIES, RankUtils.inferRank("Abies", "Abies", "alba", null, null));
034    assertEquals(Rank.SPECIES, RankUtils.inferRank("Abies", null, "alba", null, null));
035    assertEquals(Rank.INFRAGENERIC_NAME, RankUtils.inferRank(null, "Abies", null, null, null));
036    assertEquals(Rank.INFRAGENERIC_NAME, RankUtils.inferRank("", "Abies", null, null, null));
037    assertEquals(Rank.VARIETY, RankUtils.inferRank(null, "Abies", "alba", "var.", "alpina"));
038    assertEquals(Rank.PATHOVAR, RankUtils.inferRank(null, "Pseudomonas", "syringae ", "pv.", "aceris"));
039    assertEquals(Rank.SUBSPECIES, RankUtils.inferRank(null, "Abies", "alba", "ssp.", null));
040    assertEquals(Rank.SPECIES, RankUtils.inferRank(null, "Abies", null, "spec.", null));
041    assertEquals(Rank.SUPRAGENERIC_NAME, RankUtils.inferRank("Neurolaenodinae", null, null, "ib.", null));
042    assertEquals(Rank.SUPRAGENERIC_NAME, RankUtils.inferRank("Neurolaenodinae", null, null, "supersubtrib.", null));
043    assertEquals(Rank.BIOVAR, RankUtils.inferRank(null, "Pseudomonas", "syringae ", "bv.", "aceris"));
044    assertEquals(Rank.BIOVAR, RankUtils.inferRank(null, "Thymus", "vulgaris", "biovar", "geraniol"));
045    assertEquals(Rank.CHEMOFORM, RankUtils.inferRank(null, "Thymus", "vulgaris", "ct.", "geraniol"));
046    assertEquals(Rank.CHEMOFORM, RankUtils.inferRank(null, "Thymus", "vulgaris", "chemoform", "geraniol"));
047    assertEquals(Rank.CHEMOVAR, RankUtils.inferRank(null, "Thymus", "vulgaris", "chemovar", "geraniol"));
048    assertEquals(Rank.SEROVAR, RankUtils.inferRank(null, "Thymus", "vulgaris", "serovar", "geraniol"));
049
050    // should not be able to infer the correct family
051    assertEquals(Rank.UNRANKED, RankUtils.inferRank("Compositae", null, null, null, null));
052  }
053
054  @Test
055  public void testInferRank2() {
056    for (Rank r : Rank.values()) {
057      if (r.getMarker() != null) {
058        assertEquals(r.getMarker(), r, RankUtils.inferRank(r.getMarker()));
059        assertEquals(r.getMarker(), r, RankUtils.inferRank("Gagga", null, null, r.getMarker(), null));
060      }
061    }
062  }
063
064
065  @Test
066  public void testSuffixMap() {
067   
068    for (Map.Entry<String, Rank> stringRankEntry : NAMES.entrySet()) {
069      Rank r = null;
070      for (String suffix : RankUtils.SUFFICES_RANK_MAP.keySet()) {
071        if (stringRankEntry.getKey().endsWith(suffix)) {
072          r = RankUtils.SUFFICES_RANK_MAP.get(suffix);
073          break;
074        }
075      }
076      assertEquals(stringRankEntry.getValue(), r);
077    }
078  }
079
080
081  @Test
082  public void testRankMarkers() {
083    for (Rank r : Rank.values()) {
084      if (r.notOtherOrUnknown() && r != Rank.CULTIVAR_GROUP) {
085        assertEquals(r, RankUtils.inferRank(r.getMarker()));
086        assertEquals(r, RankUtils.inferRank("notho"+r.getMarker()));
087      }
088    }
089  }
090}
