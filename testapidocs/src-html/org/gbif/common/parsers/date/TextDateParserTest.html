001package org.gbif.common.parsers.date;
002
003import org.gbif.common.parsers.core.ParseResult;
004
005import java.time.LocalDate;
006import java.time.Month;
007import java.time.YearMonth;
008import java.time.temporal.TemporalAccessor;
009
010import org.junit.Test;
011
012import static junit.framework.TestCase.assertEquals;
013import static org.junit.Assert.assertFalse;
014
015/**
016 * Unit tests for {@link TextDateParser}.
017 */
018public class TextDateParserTest {
019
020  private TextDateParser TEXTDATE_PARSER = new TextDateParser();
021
022  @Test
023  public void testTextDateParsing(){
024    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("2nd jan. 2018");
025    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
026
027    parseResult = TEXTDATE_PARSER.parse("2018 january 2");
028    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
029
030    parseResult = TEXTDATE_PARSER.parse("2018-01-02");
031    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
032
033    parseResult = TEXTDATE_PARSER.parse("2018年1月2日");
034    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
035  }
036
037  @Test
038  public void testTextDateTimeParsing(){
039    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("1978-1-1T00:00");
040    assertEquals(LocalDate.of(1978, Month.JANUARY, 1), LocalDate.from(parseResult.getPayload()));
041
042    parseResult = TEXTDATE_PARSER.parse("1978-01-01T02:12:43Z");
043    assertEquals(LocalDate.of(1978, Month.JANUARY, 1), LocalDate.from(parseResult.getPayload()));
044  }
045
046  @Test
047  public void testDateParts(){
048    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("2018.0", "1.0", "02.0");
049    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
050
051    parseResult = TEXTDATE_PARSER.parse("2018.0", "jan", "02.0");
052    assertEquals(LocalDate.of(2018, Month.JANUARY, 2), LocalDate.from(parseResult.getPayload()));
053
054    parseResult = TEXTDATE_PARSER.parse("2018.0", "jan", "");
055    assertEquals(YearMonth.of(2018, Month.JANUARY), YearMonth.from(parseResult.getPayload()));
056  }
057
058  @Test
059  public void testTextDateParsingNullEmpty(){
060    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("");
061    assertFalse(parseResult.isSuccessful());
062
063    parseResult = TEXTDATE_PARSER.parse(null);
064    assertFalse(parseResult.isSuccessful());
065  }
066
067  @Test
068  public void testTextDateParsingInvalidDate(){
069    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("32nd jan. 2018");
070    assertFalse(parseResult.isSuccessful());
071
072    parseResult = TEXTDATE_PARSER.parse("2nd n/a 2018");
073    assertFalse(parseResult.isSuccessful());
074
075    parseResult = TEXTDATE_PARSER.parse("15 jan-fev 2018");
076    assertFalse(parseResult.isSuccessful());
077  }
078
079  @Test
080  public void testTextDatePartsParsing(){
081    ParseResult<TemporalAccessor> parseResult = TEXTDATE_PARSER.parse("1879", "11", "1");
082    assertEquals(LocalDate.of(1879, Month.NOVEMBER, 1), LocalDate.from(parseResult.getPayload()));
083
084    parseResult = TEXTDATE_PARSER.parse("1879", "a", "1");
085    assertFalse(parseResult.isSuccessful());
086  }
087}
