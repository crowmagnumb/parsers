084        assertEquivalent("01/01/1978", DateParseUtils.parse("1978", "01", "01"), ParseResult.CONFIDENCE.DEFINITE);

087        assertEquals(ParseResult.STATUS.FAIL, DateParseUtils.parse("-1", "1", "1").getStatus());
088        assertEquals(ParseResult.STATUS.FAIL, DateParseUtils.parse(null, null, null).getStatus());

091    @Test
092    public void testAtomizeString() {
093        String input = "21/12/2010";
094        ParseResult<YearMonthDay> result = DateParseUtils.atomize(input);
095        LOGGER.info("{} atomized to {}", input, result);
096        assertEquals(ParseResult.STATUS.SUCCESS, result.getStatus());
097        assertEquals(ParseResult.CONFIDENCE.DEFINITE, result.getConfidence());
098        assertNotNull(result.getPayload());
099        assertEquals(new YearMonthDay("2010", "12", "21"), result.getPayload());
100
101        input = "16/07/99";
102        result = DateParseUtils.atomize(input);
103        LOGGER.info("{} atomized to {}", input, result);
104        assertEquals(ParseResult.STATUS.SUCCESS, result.getStatus());
105        assertEquals(ParseResult.CONFIDENCE.DEFINITE, result.getConfidence());
106        assertNotNull(result.getPayload());
107        assertEquals(new YearMonthDay("1999", "7", "16"), result.getPayload());
108    }

165    public void testNormalizeStringStringString() {
166        assertEquals(new YearMonthDay("1978", "12", "21"), DateParseUtils.normalize("1978", "DEC", "21"));
167        assertEquals(new YearMonthDay("1850", null, null), DateParseUtils.normalize("1850", "\\N", "\\N"));
168        assertEquals(new YearMonthDay("1800", "06", null), DateParseUtils.normalize("1800", "June", "\\N"));
169        // test that the year inference works ok
170        assertEquals(new YearMonthDay("1974", "03", "12"), DateParseUtils.normalize("12/3/74", null, null));
171        assertEquals(new YearMonthDay("2004", "12", "11"), DateParseUtils.normalize("11/12/04", null, null));
172        // test that 2 digit years are handled sensibly
173        assertEquals(new YearMonthDay("1997", "08", "03"), DateParseUtils.normalize("97", "8", "3"));
174        // when we see an erroneous date, try to make most sense out of it
175        assertEquals(new YearMonthDay("1929", "07", "01"), DateParseUtils.normalize("1929-06-31", null, null));
176
177        assertEquals(new YearMonthDay("1965", "12", "02"), DateParseUtils.normalize("1965", "12", "02"));
178        assertEquals(new YearMonthDay("1992", null, null), DateParseUtils.normalize("1992", "00", "00"));
179        assertEquals(new YearMonthDay("1999", "02", "01"), DateParseUtils.normalize("1999-02-01", null, null));
180
181        assertEquals(new YearMonthDay(null, "08", "26"), DateParseUtils.normalize("32767-08-26", null, null));

225    private void assertEquivalent(String expected, ParseResult<Date> found, ParseResult.CONFIDENCE c) {
226        if (ParseResult.STATUS.ERROR == found.getStatus()) {
227            LOGGER.warn("Unexpected error found in parsing", found.getError());
228        }
229        assertEquals(ParseResult.STATUS.SUCCESS, found.getStatus());
230        SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy");
231        LOGGER.info("Expected [{}]: {}", expected, sdf.format(found.getPayload()));
232        assertEquals(c, found.getConfidence());
233        assertNotNull(found.getPayload());
234        try {
235            assertEquals(DateUtils.parseDateStrictly(expected, new String[]{"dd/MM/yyyy"}), found.getPayload());
236        } catch (ParseException e) {
237            fail("Error in the test class.  Expected date is not in the correct format of dd/MM/yyyy:" + expected);
238        }
239    }
