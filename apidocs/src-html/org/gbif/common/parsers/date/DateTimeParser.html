001package org.gbif.common.parsers.date;
002
003
004import java.time.format.DateTimeFormatter;
005import java.time.format.DateTimeParseException;
006import java.time.temporal.TemporalAccessor;
007import java.time.temporal.TemporalQuery;
008import javax.annotation.Nullable;
009import javax.validation.constraints.NotNull;
010
011import com.google.common.base.Preconditions;
012
013
014/**
015 * Adds some flexibility around {@link DateTimeFormatter} with the {@link DateTimeSeparatorNormalizer} and
016 * simple optimization the support of DateFormatHint.
017 *
018 *
019 * This class is thread-safe once an instance is created.
020 *
021 */
022public class DateTimeParser {
023
024  private final DateTimeFormatter formatter;
025  private final DateTimeSeparatorNormalizer normalizer;
026  private final DateFormatHint hint;
027
028  private final TemporalQuery<?>[] types;
029  private final int minLength;
030
031  /**
032   * Package protected constructor.
033   * Use {@link DateTimeParserBuilder}
034   *
035   * @param formatter
036   * @param normalizer optional, can be null
037   * @param hint
038   * @param type
039   * @param minLength
040   */
041  DateTimeParser(@NotNull DateTimeFormatter formatter, @Nullable DateTimeSeparatorNormalizer normalizer,
042                 @NotNull DateFormatHint hint, TemporalQuery<?>[] type, int minLength){
043
044    Preconditions.checkNotNull(formatter, "DateTimeFormatter can not be null");
045    Preconditions.checkNotNull(hint, "DateFormatHint can not be null");
046    Preconditions.checkNotNull(type, "TemporalQuery can not be null");
047    Preconditions.checkArgument(minLength > 0, "minLength must be greater than 0");
048
049    this.formatter = formatter;
050    this.hint = hint;
051    this.normalizer = normalizer;
052    this.minLength = minLength;
053    this.types = type;
054  }
055
056  public DateFormatHint getHint() {
057    return hint;
058  }
059
060  /**
061   * Parses the provided String as a TemporalAccessor if possible, otherwise returns null.
062   *
063   * This function fully support partial dates and will return the best possible date resolution based
064   * on the {@link DateFormatHint} provided.
065   *
066   * This function will not throw DateTimeParseException but returns null in case the input
067   * can not be parsed.
068   *
069   * @param input
070   * @return TemporalAccessor or null in case the input can not be parsed.
071   */
072  public TemporalAccessor parse(String input){
073
074    // return fast if minimum length is not meet
075    if(input.length() < minLength){
076      return null;
077    }
078
079    if(normalizer != null){
080      input = normalizer.normalize(input);
081    }
082
083    try {
084      if(types.length > 1) {
085        return formatter.parseBest(input, types);
086      }
087      return (TemporalAccessor)formatter.parse(input, types[0]);
088    }
089    catch (DateTimeParseException dpe){}
090    return null;
091  }
092
093}
