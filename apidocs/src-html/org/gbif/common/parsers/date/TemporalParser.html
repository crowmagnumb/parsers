001package org.gbif.common.parsers.date;
002
003import org.gbif.common.parsers.core.Parsable;
004import org.gbif.common.parsers.core.ParseResult;
005
006import java.time.temporal.TemporalAccessor;
007import javax.annotation.Nullable;
008
009
010/**
011 *
012 * Main interface for date/time parsing based.
013 *
014 */
015public interface TemporalParser extends Parsable<TemporalAccessor> {
016
017  /**
018   * Parse a date represented as a single String into a TemporalAccessor.
019   *
020   * @param input
021   * @param hint help to speed up the parsing and possibly return a better confidence
022   * @return result, never null
023   */
024  ParseResult<TemporalAccessor> parse(String input, @Nullable DateFormatHint hint);
025
026  /**
027   * Parse year, month, day strings as a TemporalAccessor.
028   *
029   * @param year numerical value of a year
030   * @param month value of the mont depending on the implementation, numerical value of a
031   *              month (starting at 1 for January) or possibly text.
032   * @param day numerical value of a day
033   * @return result, never null
034   */
035  ParseResult<TemporalAccessor> parse(@Nullable String year, @Nullable String month, @Nullable String day);
036
037  /**
038   * Parse year, month, day integers as a TemporalAccessor.
039   *
040   * @param year numerical value of a year
041   * @param month numerical value of a month (starting at 1 for January)
042   * @param day numerical value of a day
043   * @return result, never null
044   */
045  ParseResult<TemporalAccessor> parse(@Nullable Integer year, @Nullable Integer month, @Nullable Integer day);
046
047}
