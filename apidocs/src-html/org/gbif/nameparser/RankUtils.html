071  protected static final List<Rank> INFRASUBSPECIFIC_MICROBIAL_RANKS;
072  static {
073    List<Rank> microbialRanks = Lists.newArrayList();
074    for (Rank r : Rank.values()) {
075      if (r.isRestrictedToCode()== NomenclaturalCode.BACTERIAL && r.isInfraspecific()) {
076        microbialRanks.add(r);
077      }
078    }
079    INFRASUBSPECIFIC_MICROBIAL_RANKS = ImmutableList.copyOf(microbialRanks);
080  }

178  /**
179   * Map of rank markers to their respective rank enum.
180   */
181  protected static final Map<String, Rank> RANK_MARKER_MAP;
182  static {
183    Map<String, Rank> ranks = Maps.newHashMap();
184    for (Rank r : Rank.values()) {
185      if (r.getMarker() != null) {
186        ranks.put(r.getMarker().replaceAll("\\.", ""), r);
187      }
188    }
189    ranks.putAll(RANK_MARKER_MAP_SUPRAGENERIC);
190    ranks.putAll(RANK_MARKER_MAP_INFRAGENERIC);
191    ranks.putAll(RANK_MARKER_MAP_INFRASPECIFIC);
192    ranks.put("subser", SUBSERIES);
193    RANK_MARKER_MAP = ImmutableMap.copyOf(ranks);
194  }

196  /**
197   * An immutable map of name suffices to corresponding ranks across all kingdoms.
198   * To minimize wrong matches this map is sorted by suffix length with the first suffices being the longest and
199   * therefore most accurate matches.
200   * See http://www.nhm.ac.uk/hosted-sites/iczn/code/index.jsp?nfv=true&article=29
201   */
202  protected static final SortedMap<String, Rank> SUFFICES_RANK_MAP =
203      new ImmutableSortedMap.Builder<String, Rank>(Ordering.natural())

227  /**
228   * Tries its best to infer a rank from a given rank marker such as subsp.
229   *
230   * @return the inferred rank or null
231   */
232  public static Rank inferRank(@Nullable String rankMarker) {
233    if (rankMarker != null) {
234      return RANK_MARKER_MAP.get(NORMALIZE_RANK_MARKER.matcher(rankMarker.toLowerCase()).replaceAll(""));
235    }
236    return null;
237  }
238
239  /**
240   * Tries its best to infer a rank from an atomised name by just looking at the name parts ignoring any existing rank on the instance.
241   * As a final resort for higher monomials the suffices are inspected, but no attempt is made to disambiguate
242   * the 2 known homonym suffices -idae and -inae, but instead the far more widespread zoological versions are
243   * interpreted.
244   * @return the inferred rank or null if it cant be found.
245   */
246
247  public static Rank inferRank(ParsedName pn) {
248    Rank inferred = inferRank(pn.getGenusOrAbove(), pn.getInfraGeneric(), pn.getSpecificEpithet(), null, pn.getInfraSpecificEpithet());
249    if (inferred != null && inferred.notOtherOrUnknown()) {
250      return inferred;
251    }
252    return null;
253  }
254
255  public static Rank inferRank(
256      @Nullable String genusOrAbove,
257      @Nullable String infraGeneric,
258      @Nullable String specificEpithet,
259      @Nullable String rankMarker,
260      @Nullable String infraSpecificEpithet
261  ) {
262    // first try rank marker
263    Rank markerRank = inferRank(rankMarker);
264    if (markerRank != null) {
265      return markerRank;
266    }
267
268    // default if we cant find anything else
269    Rank rank = UNRANKED;
270    // detect rank based on parsed name
271    if (infraSpecificEpithet != null) {
272      // some infraspecific name
273      rank = INFRASPECIFIC_NAME;
274    } else if (specificEpithet != null) {
275      // a species
276      rank = SPECIES;
277    } else if (infraGeneric != null) {
278      // some infrageneric name
279      rank = INFRAGENERIC_NAME;
280    } else if (genusOrAbove != null) {
281      // a suprageneric name, check suffices
282      for (String suffix : SUFFICES_RANK_MAP.keySet()) {
283        if (genusOrAbove.endsWith(suffix)) {
284          rank = SUFFICES_RANK_MAP.get(suffix);
285          break;
286        }
287      }
288    }
289    return rank;
290  }
