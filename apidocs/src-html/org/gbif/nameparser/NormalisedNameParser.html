028/**
029 * Core parser class of the name parser that tries to take a clean name into its pieces by using regular expressions.
030 * It runs the actual regex matching in another thread that stops whenever the configured timeout is reached.
031 */
032public class NormalisedNameParser {
033  private static Logger LOG = LoggerFactory.getLogger(NormalisedNameParser.class);
034  private static final ExecutorService THREAD_POOL = Executors.newCachedThreadPool();
035  private final long timeout;  // max parsing time in milliseconds
036
037  public NormalisedNameParser(long timeout) {
038    this.timeout = timeout;
039  }

041  // name parsing
042  protected static final String NAME_LETTERS = "A-ZÏËÖÜÄÉÈČÁÀÆŒ";
043  protected static final String name_letters = "a-zïëöüäåéèčáàæœ";
044  protected static final String AUTHOR_LETTERS = NAME_LETTERS + "\\p{Lu}"; // upper case unicode letter, not numerical
045  // (\W is alphanum)
046  protected static final String author_letters = name_letters + "\\p{Ll}-"; // lower case unicode letter, not numerical
047  // (\W is alphanum)
048  protected static final String AUTHOR_PREFIXES =
049    "(?:[vV](?:an)(?:[ -](?:den|der) )? ?|von[ -](?:den |der |dem )?|(?:del|de|di|da)[`' _]|(?:Des|De|Di|N)[`' _]?|(?:de )?(?:la|le) |d'|D'|Mac|Mc|Le|St\\.? ?|Ou|O')";
050  protected static final String AUTHOR = "(?:" +
051                                         // author initials
052                                         "(?:" + "(?:[" + AUTHOR_LETTERS + "]{1,3}\\.?[ -]?){0,3}" +
053                                         // or full first name
054                                         "|[" + AUTHOR_LETTERS + "][" + author_letters + "?]{3,}" + " )?" +
055                                         // common prefixes
056                                         AUTHOR_PREFIXES + "?" +
057                                         // only allow v. in front of Capital Authornames - if included in AUTHOR_PREFIXES parseIgnoreAuthors fails
058                                         "(?:v\\. )?" +
059                                         // regular author name
060                                         "[" + AUTHOR_LETTERS + "]+[" + author_letters + "?]*\\.?" +
061                                         // potential double names, e.g. Solms-Laub.
062                                         "(?:(?:[- ](?:de|da|du)?[- ]?)[" + AUTHOR_LETTERS + "]+[" + author_letters
063                                         + "?]*\\.?)?" +
064                                         // common name suffices (ms=manuscript, not yet published)
065                                         "(?: ?(?:f|fil|j|jr|jun|junior|sr|sen|senior|ms)\\.?)?" +
066                                         // at last there might be 2 well known sanction authors for fungus, see POR-2454
067                                         "(?: *: *(?:Pers|Fr)\\.?)?" +
068                                         ")";
069  protected static final String AUTHOR_TEAM =
070    AUTHOR + "?(?:(?: ?ex\\.? | & | et | in |, ?|; ?|\\.)(?:" + AUTHOR + "|al\\.?))*";
071  protected static final Pattern AUTHOR_TEAM_PATTERN = Pattern.compile("^" + AUTHOR_TEAM + "$");
072  protected static final String YEAR = "[12][0-9][0-9][0-9?][abcdh?]?(?:[/-][0-9]{1,4})?";
073  // protected static final String YEAR_RANGE = YEAR+"(?: ?-? ?"+YEAR+")?";

075  private static final String NOTHO = "notho";
076  protected static final String RANK_MARKER_SPECIES =
077    "(?:"+NOTHO+")?(?:" + StringUtils.join(RankUtils.RANK_MARKER_MAP_INFRASPECIFIC.keySet(), "|") + ")\\.?";
078
079  private static final Function<Rank,String> REMOVE_RANK_MARKER = new Function<Rank, String>() {
080    @Override
081    public String apply(Rank rank) {
082      return rank.getMarker().replaceAll("\\.", "\\\\.");
083    }
084  };

086  protected static final String RANK_MARKER_MICROBIAL =
087    "(?:bv\\.|ct\\.|f\\. ?sp\\.|"
088    + StringUtils.join(Lists.transform(Lists.newArrayList(RankUtils.INFRASUBSPECIFIC_MICROBIAL_RANKS), REMOVE_RANK_MARKER
089    ), "|") + ")";

091  protected static final String EPHITHET_PREFIXES = "van|novae";
092  protected static final String GENETIC_EPHITHETS = "bacilliform|coliform|coryneform|cytoform|chemoform|biovar|serovar|genomovar|agamovar|cultivar|genotype|serotype|subtype|ribotype|isolate";
093  protected static final String EPHITHET = "(?:[0-9]+-?|[doml]')?"
094            + "(?:(?:" + EPHITHET_PREFIXES + ") [a-z])?"
095            + "[" + name_letters + "+-]{1,}(?<! d)[" + name_letters + "]"
096            // avoid epithets ending with the unallowed endings, e.g. serovar
097            + "(?<!(?:\\bex|"+GENETIC_EPHITHETS+"))(?=\\b)";
098  protected static final String MONOMIAL =
099    "[" + NAME_LETTERS + "](?:\\.|[" + name_letters + "]+)(?:-[" + NAME_LETTERS + "]?[" + name_letters + "]+)?";
100  // a pattern matching typical latin word endings. Helps identify name parts from authors
101  private static final Pattern LATIN_ENDINGS;
102  static {
103      try {
104          List<String> endings = FileUtils.streamToList(FileUtils.classpathStream("latin-endings.txt"));
105          LATIN_ENDINGS = Pattern.compile("(" + Joiner.on('|').skipNulls().join(endings) + ")$");
106      } catch (IOException e) {
107          throw new IllegalStateException("Failed to read latin-endings.txt from classpath resources", e);
108      }
109  }
110  protected static final String INFRAGENERIC =
111    "(?:" + "\\( ?([" + NAME_LETTERS + "][" + name_letters + "-]+) ?\\)" + "|" + "(" + StringUtils
112      .join(RankUtils.RANK_MARKER_MAP_INFRAGENERIC.keySet(), "|") + ")\\.? ?([" + NAME_LETTERS + "][" + name_letters + "-]+)"
113    + ")";
114
115  protected static final String RANK_MARKER_ALL = "("+NOTHO+")? *(" + StringUtils.join(RankUtils.RANK_MARKER_MAP.keySet(), "|") + ")\\.?";
116  private static final Pattern RANK_MARKER_ONLY = Pattern.compile("^" + RANK_MARKER_ALL + "$");

118  // main name matcher
119  public static final Pattern CANON_NAME_IGNORE_AUTHORS = Pattern.compile("^" +
120              // #1 genus/monomial
121              "(×?(?:\\?|" + MONOMIAL + "))" +
122              // #2 or #4 subgenus/section with #3 infrageneric rank marker
123              "(?:(?<!ceae) " + INFRAGENERIC + ")?" +
124              // catch author name prefixes just to ignore them so they dont become wrong epithets
125              "(?: " + AUTHOR_PREFIXES + ")?" +
126              // #5 species
127              "(?: (×?" + EPHITHET + "))?" +
128              // catch author name prefixes just to ignore them so they dont become wrong epithets
129              "(?: " + AUTHOR_PREFIXES + ")?" +
130              "(?:" +
131                // either directly a infraspecific epitheton or a author but then mandate rank marker
132                // #6 superfluent intermediate (subspecies) epithet in quadrinomials
133                "( " + EPHITHET + ")?" +
134                "(?:" +
135                  // anything in between
136                  "(?: .+)?" +
137                  // #7 infraspecies rank
138                  "( " + RANK_MARKER_SPECIES + "[ .])" +
139                ")?" +
140                // #8 infraspecies epithet
141                " (×?" + EPHITHET + ")" +
142              ")?" +
143              "(?: " +
144                // #9 microbial rank
145                "(" + RANK_MARKER_MICROBIAL + ")[ .]" +
146                // #10 microbial infrasubspecific epithet
147                "(\\S+)" +
148              ")?");
149
150  public static final Pattern NAME_PATTERN = Pattern.compile("^" +
151             // #1 genus/monomial
152             "(×?(?:\\?|" + MONOMIAL + "))" +
153             // #2 or #4 subgenus/section with #3 infrageneric rank marker
154             "(?:(?<!ceae) " + INFRAGENERIC + ")?" +
155             // #5 species
156             "(?: (×?" + EPHITHET + "))?" +
157
158             "(?:" +
159               // #6 superfluent intermediate (subspecies) epithet in quadrinomials
160               "( " + EPHITHET + ")??" +
161               "(?:" +
162                 // strip out intermediate, irrelevant authors
163                 "(?: .+)??" +
164                 // #7 infraspecies rank
165                 "( " + RANK_MARKER_SPECIES + ")" +
166               ")?" +
167               // #8 infraspecies epitheton
168               "(?: (×?\"?" + EPHITHET + "\"?))" +
169             ")?" +
170
171             "(?: " +
172               // #9 microbial rank
173               "(" + RANK_MARKER_MICROBIAL + ")[ .]" +
174               // #10 microbial infrasubspecific epithet
175               "(\\S+)" +
176             ")?" +
177
178             // #11 entire authorship incl basionyms and year
179             "(,?" +
180               "(?: ?\\(" +
181                 // #12 basionym authors
182                 "(" + AUTHOR_TEAM + ")?" +
183                 // #13 basionym year
184                 ",?( ?" + YEAR + ")?" +
185               "\\))?" +
186
187               // #14 authors
188               "( " + AUTHOR_TEAM + ")?" +
189               // #15 year with or without brackets
190               "(?: ?\\(?,? ?(" + YEAR + ")\\)?)?" +
191             ")" +
192             "$");

195  private class MatcherCallable implements Callable<Matcher> {
196    private final String scientificName;
197
198    MatcherCallable(String scientificName) {
199      this.scientificName = scientificName;
200    }
201
202    @Override
203    public Matcher call() throws Exception {
204      Matcher matcher = NAME_PATTERN.matcher(scientificName);
205      matcher.find();
206      return matcher;
207    }
208  }

210  /**
211   * Tries to parse a name string with the full regular expression.
212   * In very few, extreme cases names with very long authorships might cause the regex to never finish or take hours
213   * we run this parsing in a separate thread that can be stopped if it runs too long.
214   * @param cn
215   * @param scientificName
216   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
217   * @return  true if the name could be parsed, false in case of failure
218   */
219  public boolean parseNormalisedName(ParsedName cn, String scientificName, @Nullable Rank rank) {
220    LOG.debug("Parse normed name string: {}", scientificName);
221    FutureTask<Matcher> task = new FutureTask<Matcher>(new MatcherCallable(scientificName));
222    THREAD_POOL.execute(task);
223
224    try {
225      Matcher matcher = task.get(timeout, TimeUnit.MILLISECONDS);
226      // if there was no match in the callable, this can yield an IllegalStateException
227      if (matcher.group(0).equals(scientificName)) {
228        if (LOG.isDebugEnabled()) {
229          logMatcher(matcher);
230        }
231        cn.setGenusOrAbove(StringUtils.trimToNull(matcher.group(1)));
232        boolean bracketSubrankFound = false;
233        if (matcher.group(2) != null) {
234          bracketSubrankFound = true;
235          cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(2)));
236        } else if (matcher.group(4) != null) {
237          setRank(cn, matcher.group(3));
238          cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(4)));
239        }
240        cn.setSpecificEpithet(StringUtils.trimToNull(matcher.group(5)));
241        if (matcher.group(6) != null && matcher.group(6).length() > 1 && !matcher.group(6).contains("null")) {
242          // 4 parted name, so its below subspecies
243          cn.setRank(Rank.INFRASUBSPECIFIC_NAME);
244        }
245        if (matcher.group(7) != null && matcher.group(7).length() > 1) {
246          setRank(cn, matcher.group(7));
247        }
248        cn.setInfraSpecificEpithet(StringUtils.trimToNull(matcher.group(8)));
249
250        // microbial ranks
251        if (matcher.group(9) != null) {
252          setRank(cn, matcher.group(9));
253          cn.setInfraSpecificEpithet(matcher.group(10));
254        }
255
256        // #11 is entire authorship, not stored in ParsedName
257        cn.setBracketAuthorship(StringUtils.trimToNull(matcher.group(12)));
258        if (bracketSubrankFound && infragenericIsAuthor(cn, rank)) {
259          // rather an author than a infrageneric rank. Swap
260          cn.setBracketAuthorship(cn.getInfraGeneric());
261          cn.setInfraGeneric(null);
262          LOG.debug("swapped subrank with bracket author: {}", cn.getBracketAuthorship());
263        }
264        if (matcher.group(13) != null && matcher.group(13).length() > 2) {
265          String yearAsString = matcher.group(13).trim();
266          cn.setBracketYear(yearAsString);
267        }
268        cn.setAuthorship(StringUtils.trimToNull(matcher.group(14)));
269        if (matcher.group(15) != null && matcher.group(15).length() > 2) {
270          String yearAsString = matcher.group(15).trim();
271          cn.setYear(yearAsString);
272        }
273
274        // make sure (infra)specific epithet is not a rank marker!
275        lookForIrregularRankMarker(cn);
276        // 2 letter epitheta can also be author prefixes - check that programmatically, not in regex
277        checkEpithetVsAuthorPrefx(cn);
278
279        // if no rank was parsed but given externally use it!
280        if (cn.getRank() == null && rank != null) {
281            cn.setRank(rank);
282        }
283        return true;
284      }
285
286    } catch (InterruptedException e) {
287      LOG.warn("InterruptedException for name: {}", scientificName, e);
288    } catch (ExecutionException e) {
289      LOG.warn("ExecutionException for name: {}", scientificName, e);
290    } catch (IllegalStateException e) {
291      // we simply had no match
292    } catch (TimeoutException e) {
293      // timeout
294      LOG.info("Parsing timeout for name: {}", scientificName);
295    }
296
297    return false;
298  }

300  /**
301   * Sets the parsed names rank based on a found rank marker
302   * Potentially also sets the notho field in case the rank marker indicates a hybrid
303   * @param pn
304   * @param rankMarker
305   */
306  protected static void setRank(ParsedName pn, String rankMarker) {
307    rankMarker = StringUtils.trimToNull(rankMarker);
308    Rank rank = RankUtils.inferRank(rankMarker);
309    pn.setRank(rank);
310    if (rank != null && rankMarker.startsWith(NOTHO)) {
311      if (rank.isInfraspecific()) {
312        pn.setNotho(NamePart.INFRASPECIFIC);
313      } else if (rank == Rank.SPECIES) {
314        pn.setNotho(NamePart.SPECIFIC);
315      } else if (rank.isInfrageneric()) {
316        pn.setNotho(NamePart.INFRAGENERIC);
317      } else if (rank == Rank.GENUS) {
318        pn.setNotho(NamePart.GENERIC);
319      }
320    }
321  }
322
323    private static boolean infragenericIsAuthor(ParsedName pn, Rank rank) {
324        return pn.getBracketAuthorship() == null && pn.getSpecificEpithet() == null && (
325                rank != null && !(rank.isInfrageneric() && !rank.isSpeciesOrBelow())
326                        //|| pn.getInfraGeneric().contains(" ")
327                        || rank == null && !LATIN_ENDINGS.matcher(pn.getInfraGeneric()).find()
328        );
329    }

331  /**
332   *
333   * @param cn
334   * @param scientificName
335   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
336   * @return  true if the name could be parsed, false in case of failure
337   */
338  public boolean parseNormalisedNameIgnoreAuthors(ParsedName cn, String scientificName, @Nullable Rank rank) {
339    LOG.debug("Parse normed name string ignoring authors: {}", scientificName);
340
341    // match for canonical
342    Matcher matcher = CANON_NAME_IGNORE_AUTHORS.matcher(scientificName);
343    boolean matchFound = matcher.find();
344    if (matchFound) {
345      if (LOG.isDebugEnabled()) {
346        logMatcher(matcher);
347      }
348      cn.setGenusOrAbove(StringUtils.trimToNull(matcher.group(1)));
349      if (matcher.group(2) != null) {
350        // subrank in paranthesis. Not an author?
351        cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(2)));
352        if (infragenericIsAuthor(cn, rank)) {
353            // rather an author...
354            cn.setInfraGeneric(null);
355        }
356      } else if (matcher.group(4) != null) {
357        // infrageneric with rank indicator given
358        setRank(cn, matcher.group(3));
359        cn.setInfraGeneric(StringUtils.trimToNull(matcher.group(4)));
360      }
361      cn.setSpecificEpithet(StringUtils.trimToNull(matcher.group(5)));
362      if (matcher.group(6) != null && matcher.group(6).length() > 1 && !matcher.group(6).contains("null")) {
363        // 4 parted name, so its below subspecies
364        cn.setRank(Rank.INFRASUBSPECIFIC_NAME);
365      }
366      if (matcher.group(7) != null && matcher.group(7).length() > 1) {
367        setRank(cn, matcher.group(7));
368      }
369      if (matcher.group(8) != null && matcher.group(8).length() >= 2) {
370        setCanonicalInfraSpecies(cn, matcher.group(8));
371      }
372      if (matcher.group(9) != null) {
373        setRank(cn, matcher.group(9));
374        cn.setInfraSpecificEpithet(matcher.group(10));
375      }
376
377      // make sure (infra)specific epithet is not a rank marker!
378      lookForIrregularRankMarker(cn);
379
380      return true;
381    }
382    return false;
383  }
384
385  private static void setCanonicalInfraSpecies(ParsedName pn, String epi) {
386    if (epi == null || epi.equalsIgnoreCase("sec") || epi.equalsIgnoreCase("sensu")) {
387      return;
388    }
389    pn.setInfraSpecificEpithet(StringUtils.trimToNull(epi));
390  }

392  /**
393   * if no rank marker is set yet inspect epitheta for wrongly placed rank markers and modify parsed name accordingly.
394   * This is sometimes the case for informal names like: Coccyzus americanus ssp.
395   *
396   * @param cn the already parsed name
397   */
398  private void lookForIrregularRankMarker(ParsedName cn) {
399    if (cn.getRank() == null) {
400      if (cn.getInfraSpecificEpithet() != null) {
401        Matcher m = RANK_MARKER_ONLY.matcher(cn.getInfraSpecificEpithet());
402        if (m.find()) {
403          // we found a rank marker, make it one
404          setRank(cn, cn.getInfraSpecificEpithet());
405          cn.setInfraSpecificEpithet(null);
406        }
407      } else if (cn.getSpecificEpithet() != null) {
408        Matcher m = RANK_MARKER_ONLY.matcher(cn.getSpecificEpithet());
409        if (m.find()) {
410          // we found a rank marker, make it one
411          setRank(cn, cn.getSpecificEpithet());
412          cn.setSpecificEpithet(null);
413        }
414      }
415    }
416  }

418  /**
419   * 2 letter epitheta can also be author prefixes - check that programmatically, not in regex
420   */
421  private void checkEpithetVsAuthorPrefx(ParsedName cn) {
422    if (cn.getRank() == null) {
423      if (cn.getInfraSpecificEpithet() != null) {
424        // might be subspecies without rank marker
425        // or short authorship prefix in epithet. test
426        String extendedAuthor = cn.getInfraSpecificEpithet() + " " + cn.getAuthorship();
427        Matcher m = AUTHOR_TEAM_PATTERN.matcher(extendedAuthor);
428        if (m.find()) {
429          // matches author. Prefer that
430          LOG.debug("use infraspecific epithet as author prefix");
431          cn.setInfraSpecificEpithet(null);
432          cn.setAuthorship(extendedAuthor);
433        }
434      } else {
435        // might be monomial with the author prefix erroneously taken as the species epithet
436        String extendedAuthor = cn.getSpecificEpithet() + " " + cn.getAuthorship();
437        Matcher m = AUTHOR_TEAM_PATTERN.matcher(extendedAuthor);
438        if (m.find()) {
439          // matches author. Prefer that
440          LOG.debug("use specific epithet as author prefix");
441          cn.setSpecificEpithet(null);
442          cn.setAuthorship(extendedAuthor);
443        }
444      }
445    }
446  }
447
448  private void logMatcher(Matcher matcher) {
449    int i = -1;
450    while (i < matcher.groupCount()) {
451      i++;
452      LOG.debug("  {}: >{}<", i, matcher.group(i));
453    }
454  }
