196  /**
197   * Fully parse the supplied name also trying to extract authorships, a conceptual sec reference, remarks or notes
198   * on the nomenclatural status. In some cases the authorship parsing proves impossible and this nameparser will
199   * return null.
200   *
201   * For strings which are no scientific names and scientific names that cannot be expressed by the ParsedName class
202   * the parser will throw an UnparsableException with a given NameType and the original, unparsed name. This is the
203   * case for all virus names and proper hybrid formulas, so make sure you catch and process this exception.
204   *
205   * @param scientificName the full scientific name to parse
206   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
207   *
208   * @throws org.gbif.nameparser.UnparsableException
209   */
210  public ParsedName parse(final String scientificName, @Nullable Rank rank) throws UnparsableException {
211    if (Strings.isNullOrEmpty(scientificName)) {
212      throw new UnparsableException(NameType.NO_NAME, scientificName);
213    }
214    long start = 0;
215    if (LOG.isDebugEnabled()) {
216      start = System.currentTimeMillis();
217    }
218
219    ParsedName pn = new ParsedName();
220    pn.setScientificName(scientificName);
221
222    // clean name, removing seriously wrong things
223    String name = preClean(scientificName);
224
225    // remove extinct markers
226    name = EXTINCT_PATTERN.matcher(name).replaceFirst("");
227
228    // before any cleaning test for properly quoted candidate names
229    Matcher m = IS_CANDIDATUS_QUOTE_PATTERN.matcher(scientificName);
230    if (m.find()) {
231      pn.setType(NameType.CANDIDATUS);
232      name = m.replaceFirst(m.group(2));
233    }
234
235    // normalize bacterial rank markers
236    name = TYPE_TO_VAR.matcher(name).replaceAll("$1var");
237
238    // parse out species/strain names with numbers found in Genebank/EBI names, e.g. Advenella kashmirensis W13003
239    m = STRAIN.matcher(name);
240    if (m.find()) {
241      name = m.replaceFirst(m.group(1));
242      pn.setType(NameType.INFORMAL);
243      pn.setStrain(m.group(2));
244      LOG.debug("Strain: {}", m.group(2));
245    }
246
247    // remove placeholders from infragenerics and authors and set type
248    m = REMOVE_PLACEHOLDER_AUTHOR.matcher(name);
249    if (m.find()) {
250      name = m.replaceFirst(" $1");
251      pn.setType(NameType.PLACEHOLDER);
252    }
253    m = REMOVE_PLACEHOLDER_INFRAGENERIC.matcher(name);
254    if (m.find()) {
255      name = m.replaceFirst("");
256      pn.setType(NameType.PLACEHOLDER);
257    }
258
259    // detect further unparsable names
260    if (PLACEHOLDER.matcher(name).find()) {
261      throw new UnparsableException(NameType.PLACEHOLDER, scientificName);
262    }
263
264    if (IS_VIRUS_PATTERN.matcher(name).find() || IS_VIRUS_PATTERN_CASE_SENSITIVE.matcher(name).find()) {
265      throw new UnparsableException(NameType.VIRUS, scientificName);
266    }
267
268    // detect RNA/DNA gene/strain names and flag as informal
269    if (IS_GENE.matcher(name).find()) {
270      pn.setType(NameType.INFORMAL);
271    }
272
273    // normalise name
274    name = normalize(name);
275    if (Strings.isNullOrEmpty(name)) {
276      throw new UnparsableException(NameType.NO_NAME, scientificName);
277    }
278
279    // parse cultivar names first BEFORE we strongly normalize
280    // this will potentially remove quotes needed to find cultivar names
281    // this will potentially remove quotes needed to find cultivar group names
282    m = CULTIVAR_GROUP.matcher(name);
283    if (m.find()) {
284      pn.setCultivarEpithet(m.group(1));
285      name = m.replaceFirst(" ");
286      pn.setType(NameType.CULTIVAR);
287      String cgroup = m.group(2);
288      if (cgroup.equalsIgnoreCase("grex")) {
289        pn.setRank(Rank.GREX);
290      } else {
291        pn.setRank(Rank.CULTIVAR_GROUP);
292      }
293    }
294    m = CULTIVAR.matcher(name);
295    if (m.find()) {
296      pn.setCultivarEpithet(m.group(1));
297      name = m.replaceFirst(" ");
298      pn.setType(NameType.CULTIVAR);
299      pn.setRank(Rank.CULTIVAR);
300    }
301
302
303
304    // name without any latin char letter at all?
305    if (NO_LETTERS.matcher(name).find()) {
306      throw new UnparsableException(NameType.NO_NAME, scientificName);
307    }
308
309    if (HYBRID_FORMULA_PATTERN.matcher(name).find()) {
310      throw new UnparsableException(NameType.HYBRID, scientificName);
311    }
312
313    m = IS_CANDIDATUS_PATTERN.matcher(name);
314    if (m.find()) {
315      pn.setType(NameType.CANDIDATUS);
316      name = m.replaceFirst("");
317    }
318
319    // extract nom.illeg. and other nomen status notes
320    m = EXTRACT_NOMSTATUS.matcher(name);
321    if (m.find()) {
322      pn.setNomStatus(StringUtils.trimToNull(m.group(1)));
323      name = m.replaceFirst("");
324      // if there was a rank given in the nom status populate the rank marker field
325      if (pn.getNomStatus() != null) {
326        Matcher rm = NOV_RANK_MARKER.matcher(pn.getNomStatus());
327        if (rm.find()) {
328          NormalisedNameParser.setRank(pn, rm.group(1));
329        }
330      }
331    }
332
333    // extract sec reference
334    m = EXTRACT_SENSU.matcher(name);
335    if (m.find()) {
336      pn.setSensu(StringUtils.trimToNull(m.group(1)));
337      name = m.replaceFirst("");
338    }
339    // extract other remarks
340    m = EXTRACT_REMARKS.matcher(name);
341    if (m.find()) {
342      pn.setRemarks(StringUtils.trimToNull(m.group(1)));
343      name = m.replaceFirst("");
344    }
345
346    // check for indets unless we already have a cultivar
347    if (pn.getType() != NameType.CULTIVAR) {
348      m = RANK_MARKER_AT_END.matcher(name);
349      // f. is a marker for forms, but more often also found in authorships as "filius" - son of.
350      // so ignore those
351      if (m.find() && !(name.endsWith(" f.") || name.endsWith(" f"))) {
352        pn.setType(NameType.INFORMAL);
353        NormalisedNameParser.setRank(pn, m.group(2));
354        name = m.replaceAll("");
355      }
356      m = NORM_INDET.matcher(name);
357      if (m.find()) {
358        pn.setType(NameType.INFORMAL);
359        name = m.replaceAll(" ");
360      }
361
362    }
363
364    name = normalizeStrong(name);
365    if (Strings.isNullOrEmpty(name)) {
366      throw new UnparsableException(NameType.DOUBTFUL, scientificName);
367    }
368
369    // remember current rank for later reuse
370    Rank preparsingRank = pn.getRank();
371
372    // try regular parsing
373    boolean parsed = nnParser.parseNormalisedName(pn, name, rank);
374    if (!parsed) {
375      // try again with stronger normalisation, maybe that helps...
376      LOG.debug("Can't parse, use dirty normalizer");
377      final String deDirtedName = cleanStrong(name);
378      parsed = nnParser.parseNormalisedName(pn, deDirtedName, rank);
379      if (!parsed) {
380        LOG.debug("Still can't parse, try to ignore authors");
381        // try to parse canonical alone ignoring authorship as last resort
382        parsed = nnParser.parseNormalisedNameIgnoreAuthors(pn, deDirtedName, rank);
383        pn.setAuthorsParsed(false);
384        if (!parsed) {
385          // we just cant parse this one
386          // try to spot a virus name once we know its not a scientific name
387          m = IS_VIRUS_PATTERN_POSTFAIL.matcher(name);
388          if (m.find()) {
389            throw new UnparsableException(NameType.VIRUS, scientificName);
390          }
391
392          throw new UnparsableException(NameType.DOUBTFUL, scientificName);
393        }
394      }
395    }
396
397    // if we established a rank during preparsing make sure we use this not the parsed one
398    if (preparsingRank != null) {
399      pn.setRank(preparsingRank);
400    }
401
402    // primarily make sure again we haven't parsed a non name
403    postAssertParsing(pn, scientificName, name);
404
405    // determine name type if not yet assigned
406    determineNameType(pn, scientificName);
407
408    // determine rank if not yet assigned
409    if (pn.getRank() == null) {
410      pn.setRank(RankUtils.inferRank(pn));
411    }
412
413    LOG.debug("Parsing time: {}", (System.currentTimeMillis() - start));
414    return pn;
415  }

417  /**
418   * Fully parses a name using #parse(String, Rank) but converts names that throw a UnparsableException
419   * into ParsedName objects with the scientific name, rank and name type given.
420   */
421  public ParsedName parseQuietly(final String scientificName, @Nullable Rank rank) {
422    ParsedName p;
423    try {
424      p = parse(scientificName, rank);
425
426    } catch (UnparsableException e) {
427      p = new ParsedName();
428      p.setScientificName(scientificName);
429      p.setRank(rank);
430      p.setType(e.type);
431      p.setParsed(false);
432      p.setAuthorsParsed(false);
433    }
434
435    return p;
436  }

438  /**
439   * parses the name without authorship and returns the ParsedName.canonicalName() string
440   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
441   */
442  public String parseToCanonical(String scientificName, @Nullable Rank rank) {
443    if (Strings.isNullOrEmpty(scientificName)) {
444      return null;
445    }
446    try {
447      ParsedName pn = parse(scientificName, rank);
448      if (pn != null) {
449        return pn.canonicalName();
450      }
451    } catch (UnparsableException e) {
452      LOG.warn("Unparsable name " + scientificName + " >>> " + e.getMessage());
453    }
454    return null;
455  }

457  /**
458   * Tries to parses the name without authorship and returns the ParsedName.canonicalName() string
459   * For unparsable types and other UnparsableExceptions the original scientific name is returned.
460   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
461   */
462  public String parseToCanonicalOrScientificName(String scientificName, @Nullable Rank rank) {
463    if (Strings.isNullOrEmpty(scientificName)) {
464      return null;
465    }
466    try {
467      ParsedName pn = parse(scientificName, rank);
468      if (pn != null) {
469        return pn.canonicalName();
470      }
471    } catch (UnparsableException e) {
472      LOG.warn("Unparsable name " + scientificName + " >>> " + e.getMessage());
473    }
474    return StringUtils.normalizeSpace(scientificName.trim());
475  }

477  /**
478   * A very optimistic cleaning intended for names potentially very very dirty
479   *
480   * @param name To normalize
481   *
482   * @return The normalized name
483   */
484  protected static String cleanStrong(String name) {
485    if (name != null) {
486      // remove final & which causes long parse times
487
488      // test for known bad suffixes like in Palythoa texaensis author unknown
489      Matcher m = BAD_NAME_SUFFICES.matcher(name);
490      if (m.find()) {
491        name = m.replaceAll("");
492      }
493
494      // replace weird chars
495      name = NORM_WEIRD_CHARS.matcher(name).replaceAll(" ");
496      // swap comb and basionym authors if comb authorship is clearly identified by a &
497
498      // TODO: improve regex - some names take minutes to parse, so we cant use it as it is !
499      // Matcher m = COMB_BAS_AUTHOR_SWAP.matcher(name);
500      // if (m.find() && m.group(1)!=null && m.group(1).contains("&") && m.group(3)!=null ){
501      // System.out.println("SWAP:"+m.group(1)+"|"+m.group(2)+"|"+m.group(3)+"|"+m.group(4));
502      // name = m.replaceFirst(StringUtils.defaultString("("+m.group(3))+StringUtils.trimToEmpty(m.group(4))+")" +
503      // m.group(1)+StringUtils.trimToEmpty(m.group(2)));
504      // }
505      // uppercase the first letter, lowercase the rest of the first word
506      m = FIRST_WORD.matcher(name);
507      if (m.find() && m.group(2) == null) {
508        // System.out.println(m.group(1)+"|"+m.group(2)+"|"+m.group(3)+"|"+m.group(4));
509        name = m.replaceFirst(
510          StringUtils.defaultString(m.group(1)) + m.group(3).toUpperCase() + m.group(4).toLowerCase() + " ");
511      }
512
513      // normalize genus hybrid marker again
514      m = NORM_HYBRIDS_GENUS.matcher(name);
515      if (m.find()) {
516        name = m.replaceFirst("×$1");
517      }
518    }
519
520    return name;
521  }

523  /**
524   * Carefully normalizes a scientific name trying to maintain the original as close as possible.
525   * In particular the string is normalized by:
526   * - adding commas in front of years
527   * - trims whitespace around hyphens
528   * - unescapes unicode chars \\uhhhh, \\nnn, \xhh
529   * - pads whitespace around &
530   * - adds whitespace after dots following a genus abbreviation or rank marker
531   * - keeps whitespace before opening and after closing brackets
532   * - removes whitespace inside brackets
533   * - removes whitespace before commas
534   * - normalized hybrid marker to be the ascii multiplication sign
535   * - removes whitespace between hybrid marker and following name part in case it is NOT a hybrid formula
536   * - trims the string and replaces multi whitespace with single space
537   * - capitalizes all only uppercase words (authors are often found in upper case only)
538   *
539   * @param name To normalize
540   *
541   * @return The normalized name
542   */
543  public static String normalize(String name) {
544    if (name == null) {
545      return null;
546    }
547    name = org.gbif.utils.text.StringUtils.unescapeUnicodeChars(name);
548
549    // normalise usage of rank marker with 2 dots, i.e. forma specialis and sensu latu
550    Matcher m = FORM_SPECIALIS.matcher(name);
551    if (m.find()) {
552      name = m.replaceAll("fsp");
553    }
554    m = SENSU_LATU.matcher(name);
555    if (m.find()) {
556      name = m.replaceAll("sl");
557    }
558
559    // normalise usage of dots after abbreviated genus and rank marker
560    m = NORM_DOTS.matcher(name);
561    if (m.find()) {
562      name = m.replaceAll("$1. ");
563    }
564    // use commas before years
565    // ICZN §22A.2 http://www.iczn.org/iczn/includes/page.jsp?article=22&nfv=
566    m = COMMA_BEFORE_YEAR.matcher(name);
567    if (m.find()) {
568      name = m.replaceAll("$1, $2");
569    }
570    // no whitespace around hyphens
571    name = NORM_HYPHENS.matcher(name).replaceAll("-");
572    // use whitespace with &
573    name = NORM_AMPERSAND_WS.matcher(name).replaceAll(" & ");
574
575    // whitespace before and after brackets, keeping the bracket style
576    m = NORM_BRACKETS_OPEN.matcher(name);
577    if (m.find()) {
578      name = m.replaceAll(" $1");
579    }
580    m = NORM_BRACKETS_CLOSE.matcher(name);
581    if (m.find()) {
582      name = m.replaceAll("$1 ");
583    }
584    // remove whitespace before commas and replace double commas with one
585    m = NORM_COMMAS.matcher(name);
586    if (m.find()) {
587      name = m.replaceAll(", ");
588    }
589    // normalize hybrid markers
590    m = NORM_HYBRIDS_GENUS.matcher(name);
591    if (m.find()) {
592      name = m.replaceFirst("×$1");
593    }
594    m = NORM_HYBRIDS_EPITH.matcher(name);
595    if (m.find()) {
596      name = m.replaceFirst("$1 ×$2");
597    }
598    m = NORM_HYBRIDS_FORM.matcher(name);
599    if (m.find()) {
600      name = m.replaceAll(" × ");
601    }
602    // capitalize all entire upper case words
603    m = NORM_UPPERCASE_WORDS.matcher(name);
604    while (m.find()) {
605      name = name.replaceFirst(m.group(0), m.group(1) + m.group(2).toLowerCase());
606    }
607
608    // finally whitespace and trimming
609    name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
610    return StringUtils.trimToEmpty(name);
611  }

613  /**
614   * Does the same as a normalize and additionally removes all ( ) and "und" etc
615   * Checks if a name starts with a blacklisted name part like "Undetermined" or "Uncertain" and only returns the
616   * blacklisted word in that case
617   * so its easy to catch names with blacklisted name parts.
618   *
619   * @param name To normalize
620   *
621   * @return The normalized name
622   */
623  @VisibleForTesting
624  static String normalizeStrong(String name) {
625    if (name == null) {
626      return null;
627    }
628    // normalize all quotes to single "
629    name = NORM_QUOTES.matcher(name).replaceAll("'");
630    // enclosing quotes
631    name = REPLACE_QUOTES.matcher(name).replaceAll("");
632    // no question marks after words (after years they should remain!)
633    Matcher m = NO_Q_MARKS.matcher(name);
634    if (m.find()) {
635      name = m.replaceAll("$1");
636    }
637
638    // remove prefixes
639    name = NORM_PREFIXES.matcher(name).replaceAll("");
640
641    // remove brackets inside the genus, the kind taxon finder produces
642    m = NORM_TF_GENUS.matcher(name);
643    if (m.find()) {
644      name = m.replaceAll("$1$2 ");
645    }
646    // replace imprint years. See ICZN §22A.2.3 http://www.iczn.org/iczn/index.jsp?nfv=&article=22
647    // Ctenotus alacer Storr, 1970 ["1969"] -> Ctenotus alacer Storr, 1970
648    // C. Flahault 1887 ("1886-1888") -> C. Flahault 1887
649    m = NORM_IMPRINT_YEAR.matcher(name);
650    if (m.find()) {
651      // System.out.println(m.group(0));
652      // System.out.println(m.group(1));
653      name = m.replaceAll("$1");
654    }
655
656    // replace bibliographic in authorship
657    name = NORM_IN_COMMA.matcher(name).replaceFirst(" in ");
658    /*
659     * m = NORM_IN_BIB.matcher(name);
660     * if (m.find()) {
661     * // keep year if it only exists in IN reference
662     * Matcher mIN = EXTRACT_YEAR.matcher(m.group(0));
663     * name = m.replaceAll("");
664     * Matcher mNAME = EXTRACT_YEAR.matcher(name);
665     * if (mIN.find() && !mNAME.find()) {
666     * name = name + ", " + mIN.group(1);
667     * }
668     * }
669     */
670
671    // This is redundant, as it is done in the regular normalize function already
672    // BUT somehow all upper case authors slow down parsing so much that it can even come to an hold in the next step
673    // so we pay the price and do it twice
674    // capitalize all entire upper case words
675    m = NORM_UPPERCASE_WORDS.matcher(name);
676    while (m.find()) {
677      name = name.replaceFirst(m.group(0), m.group(1) + m.group(2).toLowerCase());
678    }
679
680    /*
681     * ICBN §46.2, Note 1.
682     * When authorship of a name differs from authorship of the publication in which it was validly published, both are
683     * sometimes cited,
684     * connected by the word "in". In such a case, "in" and what follows are part of a bibliographic citation and are
685     * better omitted
686     * unless the place of publication is being cited.
687     */
688
689    // normalise original name authorship, putting author AND year in brackets
690    // with long authorships this gets slow. Test if year exists first:
691    m = EXTRACT_YEAR.matcher(name);
692    if (m.find() && name.length() < 80) {
693      m = NORM_ORIG_AUTH.matcher(name);
694      if (m.find()) {
695        name = m.replaceAll("($1 $2)");
696      }
697      m = NORM_ORIG_AUTH2.matcher(name);
698      if (m.find()) {
699        name = m.replaceAll("($1 $2)");
700      }
701    }
702
703    // replace square brackets, keeping content (or better remove all within?)
704    name = NORM_NO_SQUARE_BRACKETS.matcher(name).replaceAll(" $1 ");
705    // replace different kind of brackets with ()
706    name = NORM_BRACKETS_OPEN_STRONG.matcher(name).replaceAll(" (");
707    name = NORM_BRACKETS_CLOSE_STRONG.matcher(name).replaceAll(") ");
708    // normalise different usages of ampersand, and, et &amp;
709    name = NORM_AND.matcher(name).replaceAll(" & ");
710    // but keep "et al." instead of "& al."
711    name = NORM_ET_AL.matcher(name).replaceAll(" et al.");
712
713    // // add commas between authors in space delimited list
714    // m = NORM_AUTH_DELIMIT.matcher(name);
715    // if (m.find()){
716    // name = m.replaceAll("$1, $2");
717    // }
718    // Bryozoan indet. 1
719    // Bryozoa sp. 2
720    // Bryozoa sp. E
721
722    name = NORM_SUFFIXES.matcher(name).replaceAll("");
723
724    // add parenthesis around subgenus if missing
725    m = NORM_SUBGENUS.matcher(name);
726    if (m.find()) {
727      name = m.replaceAll("$1 ($2) $3");
728    }
729
730    // finally whitespace and trimming
731    name = normalize(name);
732    // name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
733    return StringUtils.trimToEmpty(name);
734  }

736  /**
737   * basic careful cleaning, trying to preserve all parsable name parts
738   */
739  @VisibleForTesting
740  static String preClean(String name) {
741    // unescape unicode
742    name = org.gbif.utils.text.StringUtils.unescapeUnicodeChars(name);
743    // remove bad whitespace in html entities
744    Matcher m = XML_ENTITY_STRIP.matcher(name);
745    if (m.find()) {
746      name = m.replaceAll("&$1;");
747    }
748    // unescape html entities
749    name = StringEscapeUtils.unescapeHtml4(name);
750    // finally remove still existing bad ampersands missing the closing ;
751    name = AMPERSAND_ENTITY.matcher(name).replaceAll("& ");
752    // replace xml tags
753    name = XML_TAGS.matcher(name).replaceAll("");
754    // trim
755    name = name.trim();
756    // remove quotes in beginning and matching ones at the end
757    for (char c : QUOTES) {
758      int idx = 0;
759      while (idx < name.length() && (c == name.charAt(idx) || Character.isWhitespace(name.charAt(idx)))) {
760        idx++;
761      }
762      if (idx > 0) {
763        // check if we also find this char at the end
764        int end = 0;
765        while (c == name.charAt(name.length() - 1 - end) && (name.length() - idx - end) > 0) {
766          end++;
767        }
768        name = name.substring(idx, name.length() - end);
769      }
770    }
771    name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
772    return StringUtils.trimToEmpty(name);
773  }

775  private void determineNameType(ParsedName pn, String scientificName) {
776    if (pn.getType() == null) {
777      // a placeholder spithet only?
778      if (pn.getGenusOrAbove().equals("?")) {
779        pn.setType(NameType.PLACEHOLDER);
780      } else {
781        // a doubtful name?
782        Matcher m = DOUBTFUL.matcher(scientificName);
783        if (!m.find()) {
784          pn.setType(NameType.DOUBTFUL);
785        } else {
786          m = DOUBTFUL2.matcher(scientificName);
787          if (m.find()) {
788            pn.setType(NameType.DOUBTFUL);
789          } else {
790            pn.setType(NameType.SCIENTIFIC);
791          }
792        }
793      }
794    }
795  }

797  private void postAssertParsing(ParsedName pn, final String rawName, final String normedName) throws UnparsableException {
798    // if we only match a monomial in the 3rd pass its suspicious
799    if (pn.getGenusOrAbove() != null && !pn.isBinomial() && Character.isLowerCase(normedName.charAt(0))) {
800      // a monomial match, but it was a lower case name - doubtful at least!
801      throw new UnparsableException(NameType.DOUBTFUL, rawName);
802
803    } else if (pn.getRank() != null) {
804      if (pn.getRank().equals(Rank.CULTIVAR) && pn.getCultivarEpithet() == null) {
805        pn.setType(NameType.INFORMAL);
806
807      } else if (pn.getRank().isSpeciesOrBelow() && pn.getRank().isRestrictedToCode()!= NomenclaturalCode.CULTIVARS && !pn.isBinomial()) {
808        pn.setType(NameType.INFORMAL);
809
810      } else if (pn.getRank().isInfraspecific() && pn.getRank().isRestrictedToCode()!= NomenclaturalCode.CULTIVARS && pn.getInfraSpecificEpithet() == null) {
811        pn.setType(NameType.INFORMAL);
812
813      } else if (!pn.getRank().isSpeciesOrBelow() && pn.isBinomial()) {
814        pn.setType(NameType.DOUBTFUL);
815      }
816    }
817  }
